\begin{longtable}{|>{\ttfamily}l|L|}
\caption{Команды выгрузки данных из~стека FPU}
\label{tab:fpu-fst-list}
% \small
% \def\normalsize{\small}
\\
\hline
\normalfont\thead{Команда} & \thead{Действие}  %& \thead{Флаги} 
\\\hline

fst dest & %Копирует вещественное число из вершины стека st(0)  в~dest
Копирует $st(0)$ в~$dest$ 
\\
fstp dest & %Выталкивает вещественное число с~вершины стека st(0)  в~dest
Выталкивает $st(0)$ в~$dest$ 

Приёмник $dest$ может быть переменной в~памяти или пустым регистром сопроцессора

Если %приёмник 
$dest$ в~памяти, его размер определяется суффиксом команды \mbox{(по умолчанию "--- 32 бита, $float$)}
\\\hline



\multicolumn{2}{|c|}{\textbf{Выгрузка с~округлением}}\\\hline

fist dmem & %Копирует целое число из вершины стека  в~$dmem$
Копирует $st(0)$ в~$dmem$ как целое
\\
fistp dmem & %Выталкивает целое число с~вершины стека  в~$dmem$
Выталкивает $st(0)$ в~$dmem$ как целое

% Операнд dmem может быть только переменной в~памяти

Размер приёмника определяется суффиксом команды \mbox{(по умолчанию "--- 16 бит, $short$)}
\\\hline


fbst dmem & %Копирует целое двоично-десятичное число  из вершины стека  в~dmem
Копирует $st(0)$ в~$dmem$ как  двоично-десятичное целое
\\
fbstp dmem & %Выталкивает целое двоично-десятичное число  с~вершины стека в~dmem
Выталкивает $st(0)$ в~$dmem$ как  двоично-десятичное целое

% Операнд dmem может быть только переменной в~памяти, 
\mbox{Размер приёмника "--- 80 бит}
\\\hline

\end{longtable}
